go 读写文件需要用到 `os`  库，主要用到以下几种方法：

1.  func Create 
```go
func Create(name string)(file *File,err error)
```

  创建一个名为name的文件，如果这个文件存在那么贼会被截断（直接把旧的内容清空）。如果成功则返回打开好的文件句柄。如果失败则会返回错误。

2. func Open
 ```go
 func Open(name string)(file *File,err error)
```

  打开一个名为name的文件，如果操作成功返回的文件对象可用于读取数据，失败则返回错误。

 3. func (*File) WriteString
```go
func (f *File) WriteString(s string)(ret int, err error)
```

 这是一个`*File`接受者的方法，接受一个字符串作为参数，返回成功写入的字节数`ret`， 可以通过和比较`ret` 和 `len(s)` 来判断是否写入成功。

```go
package main

import (
	
	"fmt"
	
	"io"
	
	"os"

)

  

func main() {

	// 写入文件

	err := writeFile("test.txt", "hello world\n")

	if err != nil {

		fmt.Println("写入失败:", err)
		
		return

	}

	fmt.Println("文件写入成功！")

  

	// 读取文件
	
	content, err := readFile("test.txt") // ✅ 文件名一致
	
	if err != nil {
	
		fmt.Println("读取失败:", err)
		
		return
	
	}

	fmt.Println("文件内容:", content)

}

  

// readFile 读取文件并返回内容和可能的错误

func readFile(fileName string) (string, error) {

	file, err := os.Open(fileName)

	if err != nil {
	
		return "", err // 返回错误而不是只打印
	
	}
	
	defer file.Close() //函数调用后关闭文件
	
	data, err := io.ReadAll(file)
	
	if err != nil {
	
		return "", err
	
	}
	
	return string(data), nil // 成功时返回 nil 错误

}

  

// writeFile 写入文件

func writeFile(fileName string, content string) error {

	file, err := os.Create(fileName)
	
	if err != nil {
	
		return err
	
	}
	
	defer file.Close() //关闭文件
	
	_, err = file.WriteString(content)
	
	return err // 直接返回错误（可能是 nil）

}
```
